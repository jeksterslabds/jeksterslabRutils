Run test

sample <- util_lapply(
  FUN = FUN,
  args = args,
  par = par,
  ncores = ncores
)
sample_means <- unlist(
  util_lapply(
    FUN = mean,
    args = c(x = sample),
    par = par,
    ncores = ncores
  )
)
mean_of_means <- mean(sample_means)
mean_of_ns <- mean(
  lengths(sample)
)

test_that("sample size used is correct", {
  expect_equivalent(
    mean_of_ns,
    n
  )
})
test_that("the mean of means converges to the population mean", {
  expect_equivalent(
    round(
      x = mean_of_means,
      digits = 0
    ),
    mu,
  )
